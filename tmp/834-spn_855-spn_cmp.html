320void mtob(int metric, int *binary, int digit){     //无返回值，转好的二进制放在第二个参数binary数组里
321    int buffer[100] = {0};
322    //定义了一个buffer数组，用来收集短除法除得的余数，栈倒序输出。
323    int b, k=0;
324    for (int i = 0; i<digit; i++) {     //先把原先数据置零
325        binary[i] = 0;
326    }
327    //    unsigned int umetric = (unsigned int)metric;
328    while(metric != 0){     //短除法
329        b = metric/2;
330        buffer[k] = metric - b*2;
331        k++;
332        metric /= 2;
333    };
334    for (int i=k-1, j=digit-k;i>=0;i--, j++){     //把倒序换为返回值binary里的正序
335        binary[j] = buffer[i];
336    }

350    int k = (4*round -3) %32 +15;       //+15保证ki取到的是低十六位
351    int temp=key>>(32-k);
352    key=key<<k;
353    key=key|temp;       //循环左移
354    unsigned short ki = key;
355    //    if(round == 5) printf("%x %x\n",key,ki);
356    unsigned short pmetric = btom(p, 16);
357    pmetric = pmetric ^ ki;     //异或
358    mtob(pmetric, p, 16);
