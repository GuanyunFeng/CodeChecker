273void ModRepeatSquare(BIGNUM *Result, const BIGNUM *b, const BIGNUM *Pow, const BIGNUM *Mod)
274{
275        BIGNUM *DivRes = BN_new();
276        BIGNUM *Cup_b = BN_new();

291 
292        do {
293                BN_div(DivRes, Rem, Cup_Pow, Two, ctx);

374void Montgomerie(BIGNUM *Result, const BIGNUM *Base, const BIGNUM *Pow, const BIGNUM *Mod)
375{
376        BIGNUM *Res = BN_new();
377        BIGNUM *Two = BN_new();
378        BIGNUM *Zero = BN_new();
379        BIGNUM *Cup = BN_new();

381        BN_CTX *ctx = BN_CTX_new();
382 
383        BN_one(Result);
384        BN_set_word(Zero, 0);
385        BN_set_word(Two, 2);
386        BN_copy(Res, Base);
387        BN_copy(Pow_backup, Pow);
388 
389        while (BN_cmp(Pow_backup, Zero) == 1)
390        {
391                BN_mod(Cup, Pow_backup, Two, ctx);
392 
393                if (BN_is_zero(Cup)) {
394                        BN_mod_sqr(Cup, Res, Mod, ctx);
395                        BN_copy(Res, Cup);

398                        BN_copy(Pow_backup, Cup);
399 
400                }
401                else {
402                        BN_mod_mul(Cup, Result, Res, Mod, ctx);
403                        BN_copy(Result, Cup);
404 
405                        BN_sub(Cup, Pow_backup, BN_value_one());
406                        BN_copy(Pow_backup, Cup);
407                }
408        }
409 
410        BN_CTX_free(ctx);
